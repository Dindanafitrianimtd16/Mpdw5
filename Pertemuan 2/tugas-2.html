Penanganan Autokorelasi Metode Hildreth-Lu

library(HoRM)
# Hildreth-Lu
hildreth.lu.func<- function(r, model.rlb){
  X <- model.matrix(model.rlb)[,-1]
  y1 <- model.response(model.frame(model.rlb))
  n <- length(y1)
  t <- 2:n
  y1 <- y1[t]-r*y1[t-1]
  X <- X[t]-r*X[t-1]
  
  
  return(lm(y1~X))
}
#Pencarian rho yang meminimumkan SSE
